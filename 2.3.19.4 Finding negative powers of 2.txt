#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    double value=1;
    int n;
    
    
    cout<<"Enter value of n power: ";
    cin>>n;
    
    for(int i=1; i<=n; i++)
    {
        value *=2;
    }
    cout.precision(20);
    cout<<"Output: "<<1/value;
    return 0;
}
